/*
Question: What skills are required for the top-paying data analyst jobs?
- Use the top 10 highest-paying Data Analyst jobs from first query
- Add the specific skills required for these roles
- Why? It provides a detailed look at which-paying jobs demand certain skills,
    helping job seekers understand which skills to develop that align with top salaries
*/



WITH top_paying_jobs AS (

    SELECT  
        j.job_id,
        c.name AS company_name,
        j.job_title,
        j.salary_year_avg,
        j.job_posted_date
    FROM
        job_postings_fact AS j
    LEFT JOIN
        company_dim AS c
    ON 
        j.company_id = c.company_id
    WHERE
        job_title_short = 'Data Analyst' AND
        job_location = 'Anywhere' AND
        salary_year_avg IS NOT NULL
    ORDER BY 
        salary_year_avg DESC
    
)
SELECT
    t.job_id,
    t.company_name,
    t.job_title,
    t.salary_year_avg,
    t.job_posted_date,
    STRING_AGG(sd.skills, ',') AS skills_list
FROM top_paying_jobs AS t
INNER JOIN skills_job_dim AS sjd ON t.job_id = sjd.job_id
INNER JOIN skills_dim AS sd ON sjd.skill_id = sd.skill_id
GROUP BY 
    t.job_id,
    t.company_name,
    t.job_title,
    t.salary_year_avg,
    t.job_posted_date

ORDER BY
    salary_year_avg DESC
